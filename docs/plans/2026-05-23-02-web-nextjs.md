# Web (Next.js v1) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use `superpowers:executing-plans` to implement this plan task-by-task.

**Goal:** Ship the seven Cardstock-designed screens (Login, Home/Shelf, Rate, Recommendations, Comparison Lens, Game detail, Profile/Settings) on top of the existing Next.js 16 codebase. End state: a signed-in user can connect their BGG account to sync owned + wishlist, add games manually, rate them on a 5-point scale, see profile-mode recommendations, and overlay-compare any two games on the 12-axis radar. BGG-imported games we don't have editorial scores for appear on the shelf with a "pending scoring" badge and are excluded from the recommender.

**Architecture:** Next.js 16 App Router with `/web` features layered on top of the existing public SEO routes. Cardstock design tokens replace the current brand palette in `globals.css`. Auth via Supabase. Recommender: server-side candidate selection via the RPCs from plan 01, client-side re-rank using the existing `src/lib/scoring/` engine.

**Tech Stack:** Next.js 16, React 19, TypeScript 5, Tailwind 4, `@supabase/ssr`, `@supabase/supabase-js`, PostHog JS, Sentry Next.js SDK.

---

## Scope cuts (locked from coordination plan)

1. **No password reset flow in v1.** Magic-link covers the recovery path.
2. **No admin route in v1.** Catalog scoring stays in the existing Claude-Code-driven editorial workflow (`docs/scoring-handoff.md`). No `/admin/score-review`, no in-app score editing.
3. **BGG sync is feature-flagged.** Enabled when `BGG_API_KEY` is set in Supabase secrets. Until then, the Profile field shows "Connect BGG (coming soon)" and the Shelf header omits the re-sync button. Manual-add affordance works regardless.

If you disagree with any of these, raise before Task 1 begins.

```tsx
// src/components/auth/login-form.tsx
"use client";
import { useState } from "react";
import { createBrowserClient } from "@/lib/supabase/client";
import { Btn } from "@/components/ui";

export function LoginForm() {
  const supabase = createBrowserClient();
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [err, setErr] = useState<string | null>(null);

  async function signIn(e: React.FormEvent) {
    e.preventDefault();
    setErr(null);
    const { error } = await supabase.auth.signInWithPassword({ email, password });
    if (error) setErr(error.message);
    else window.location.href = "/shelf";
  }

  async function magicLink() {
    setErr(null);
    const { error } = await supabase.auth.signInWithOtp({
      email,
      options: { emailRedirectTo: `${location.origin}/auth/callback` },
    });
    if (error) setErr(error.message);
    else setErr("Check your inbox for the link.");
  }

  async function google() {
    await supabase.auth.signInWithOAuth({
      provider: "google",
      options: { redirectTo: `${location.origin}/auth/callback` },
    });
  }

  return (
    <form onSubmit={signIn} className="flex flex-col gap-4">
      <h2 className="font-display text-3xl">Welcome back.</h2>
      <input type="email" required placeholder="email"
        value={email} onChange={(e) => setEmail(e.target.value)}
        className="bg-paper-warm border border-ink/15 rounded-md px-3 py-2" />
      <input type="password" placeholder="password"
        value={password} onChange={(e) => setPassword(e.target.value)}
        className="bg-paper-warm border border-ink/15 rounded-md px-3 py-2" />
      {err && <p className="text-negative font-mono text-xs">{err}</p>}
      <Btn type="submit" tone="primary" size="lg">Sign in →</Btn>
      <div className="text-center text-muted font-mono text-xs uppercase tracking-wider">— or —</div>
      <Btn type="button" tone="ghost" onClick={magicLink}>Email me a magic link</Btn>
      <Btn type="button" tone="ghost" onClick={google}>Continue with Google</Btn>
      <Btn type="button" tone="ghost" disabled title="Available with iOS app">Continue with Apple</Btn>

}

// src/components/shelf/bgg-sync-button.tsx
"use client";
import { useState } from "react";
import { createBrowserClient } from "@/lib/supabase/client";
import { Btn } from "@/components/ui";
import { posthog } from "@/lib/analytics/posthog";

export function BggSyncButton({ bggUsernameSet }: { bggUsernameSet: boolean }) {
  const supabase = createBrowserClient();
  const [state, setState] = useState<"idle" | "syncing" | "ok" | "error">("idle");
  const [msg, setMsg] = useState<string | null>(null);

  async function go() {
    if (!bggUsernameSet) { window.location.href = "/profile#bgg"; return; }
    setState("syncing"); setMsg("Fetching from BGG…");
    const { data, error } = await supabase.functions.invoke("bgg-sync", { body: {} });
    if (error) { setState("error"); setMsg(error.message); return; }
    setState("ok");
    setMsg(`Imported ${data.owned} owned · ${data.wishlist} wishlist · ${data.new_unscored} pending scoring`);
    posthog?.capture("bgg_sync_completed", data);
    setTimeout(() => location.reload(), 1200);
  }

  return (
    <div className="flex items-center gap-3">
      {msg && <span className="font-mono text-xs text-muted">{msg}</span>}
      <Btn tone="primary" onClick={go} disabled={state === "syncing"}>
        {state === "syncing" ? "Syncing…" : "↻ Re-sync BGG"}
      </Btn>
    </div>
  );
}
```

useEffect(() => {

    const map: Record<string, number> = { f: 5, d: 4, s: 3, a: 2, q: 1 };

async function submitRating(value: 1 | 2 | 3 | 4 | 5) {
  const item = queue[0];
  setQueue((q) => q.slice(1));
  await supabase.from("collection_items").update({ user_rating: value }).eq("id", item.id);
  // PostHog event
  posthog?.capture("rating_submitted", { game_id: item.game.id, rating: value });
}

**Step 1: Provider**

```tsx
"use client";
import { useEffect } from "react";

export function ThemeProvider({ theme, children }: { theme: "light" | "dark" | "auto"; children: React.ReactNode }) {
  useEffect(() => {
    const apply = (mode: "light" | "dark") => {
      document.documentElement.dataset.theme = mode;
    };
    if (theme === "auto") {
      const mq = window.matchMedia("(prefers-color-scheme: dark)");
      apply(mq.matches ? "dark" : "light");
      const onChange = (e: MediaQueryListEvent) => apply(e.matches ? "dark" : "light");
      mq.addEventListener("change", onChange);
      return () => mq.removeEventListener("change", onChange);
    } else {
      apply(theme);
    }
  }, [theme]);
  return <>{children}</>;
}
```
